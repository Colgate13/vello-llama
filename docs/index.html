Installation

01

$ git clone https://github.com/Colgate13/vello-llama.git $ cd vello-llama
02

$ ./vello-installer install
03

$ vello recommend chat $ vello install qwen3-8b
04

$ vello up $ vello health # is the API up?
05

$ xdg-open http://localhost:3000

Community catalogs

default

8–24 GB VRAM 

# grab a catalog file from someone you trust
curl -O https://example.com/their-catalog.toml

# add it — vello validates the schema and copies it to catalogs/user/
vello catalog add ./their-catalog.toml

# list loaded catalogs / browse merged models
vello catalog list
vello list

# remove later by name
vello catalog remove their-catalog

# my-catalog.toml
name = "my-catalog"
version = "1"

[[model]]
id          = "qwen3-8b"
family      = "qwen3"
params_b    = 8.0
context     = 32768
tags        = ["chat", "tools", "multilingual"]
description = "Qwen 3 8B. Strong tool calling."

  [model.files]
  repo    = "bartowski/Qwen3-8B-GGUF"
  pattern = "Qwen3-8B-{quant}.gguf"
  quants  = ["Q4_K_M", "Q5_K_M", "Q6_K"]

  [model.runtime]
  ctx_size = 8192

Connecting clients

from openai import OpenAI

client = OpenAI(
    base_url="http://localhost:8080/v1",
    api_key="local",  # any string works
)

resp = client.chat.completions.create(
    model="qwen3-8b-local",
    messages=[{"role": "user", "content": "Hi"}],
)
print(resp.choices[0].message.content)
import OpenAI from "openai";

const client = new OpenAI({
  baseURL: "http://localhost:8080/v1",
  apiKey:  "local",
});

const resp = await client.chat.completions.create({
  model:    "qwen3-8b-local",
  messages: [{ role: "user", content: "Hi" }],
});
console.log(resp.choices[0].message.content);
curl http://localhost:8080/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '{
    "model": "qwen3-8b-local",
    "messages": [{"role": "user", "content": "Hi"}]
  }'
